## Create your first ASCII art SVG

This guide will walk you through creating a simple ASCII art SVG using Svelte ASCII Art. You'll learn the basics of rendering ASCII art and applying visual enhancements.

<Note>
Make sure you've [installed svelte-asciiart](/installation) before following this guide.
</Note>

## Step 1: Basic rendering

Create a new Svelte component and import the `AsciiArt` component:

```svelte +page.svelte
<script>
  import { AsciiArt } from 'svelte-asciiart';

  const text = `+----------+
|  Hello   |
|  World!  |
+----------+`;
</script>

<AsciiArt {text} />
```

This renders your ASCII art as a scalable SVG. The component automatically calculates the correct viewBox dimensions based on your text content.

<Tip>
ASCII art is rendered character-by-character. Each character becomes an SVG `<tspan>` element positioned on a precise coordinate grid.
</Tip>

## Step 2: Add a grid overlay

Enable the grid to visualize the character grid:

```svelte
<AsciiArt {text} grid gridClass="ascii-grid" />

<style>
  :global(.ascii-grid) {
    stroke: #90ee90;
    stroke-width: 0.03;
    opacity: 0.5;
  }
</style>
```

The grid spans the entire viewBox and helps you visualize how characters align on the coordinate system.

## Step 3: Add a frame and margin

Add a frame border with margin spacing:

```svelte
<AsciiArt 
  {text} 
  grid 
  frame 
  margin={[1, 2]} 
  gridClass="ascii-grid" 
  frameClass="ascii-frame" 
/>

<style>
  :global(.ascii-grid) {
    stroke: #90ee90;
    stroke-width: 0.03;
    opacity: 0.5;
  }
  
  :global(.ascii-frame) {
    stroke: #ffb366;
    stroke-width: 0.05;
  }
</style>
```

The `margin` prop accepts:
- A single number: `margin={2}` (uniform margin)
- Two numbers: `margin={[1, 2]}` (vertical, horizontal)
- Four numbers: `margin={[1, 2, 1, 2]}` (top, right, bottom, left)

Margin values are in grid cells, making it intuitive to work with character-based layouts.

## Step 4: Customize the cell aspect ratio

Adjust the character cell width-to-height ratio:

```svelte
<AsciiArt 
  {text} 
  grid 
  frame 
  margin={[1, 2]} 
  cellAspect={0.6}
  gridClass="ascii-grid" 
  frameClass="ascii-frame" 
/>
```

The default `cellAspect` is `0.6`, which matches most monospace fonts. Adjust this value if your ASCII art looks stretched or compressed:

- `0.5` - Narrower characters
- `0.6` - Default, works well with most fonts
- `0.8` - Wider characters

## Step 5: Export your SVG

Bind the SVG element to access export functions:

```svelte
<script>
  import { AsciiArt, exportSvg, exportSvgToPng } from 'svelte-asciiart';

  const text = `+----------+
|  Hello   |
|  World!  |
+----------+`;

  let svg = $state();

  async function downloadSvg() {
    if (!svg) return;
    
    const markup = exportSvg(svg, {
      includeBackground: true,
      backgroundColor: '#f3f4f6'
    });
    
    const blob = new Blob([markup], { type: 'image/svg+xml' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'ascii-art.svg';
    a.click();
    URL.revokeObjectURL(url);
  }

  async function downloadPng() {
    if (!svg) return;
    
    const dataUrl = await exportSvgToPng(svg, {
      includeBackground: true,
      backgroundColor: '#f3f4f6',
      scale: 2
    });
    
    const a = document.createElement('a');
    a.href = dataUrl;
    a.download = 'ascii-art.png';
    a.click();
  }
</script>

<AsciiArt 
  bind:svg
  {text} 
  grid 
  frame 
  margin={[1, 2]} 
  gridClass="ascii-grid" 
  frameClass="ascii-frame" 
/>

<button onclick={downloadSvg}>Download SVG</button>
<button onclick={downloadPng}>Download PNG</button>

<style>
  :global(.ascii-grid) {
    stroke: #90ee90;
    stroke-width: 0.03;
    opacity: 0.5;
  }
  
  :global(.ascii-frame) {
    stroke: #ffb366;
    stroke-width: 0.05;
  }
</style>
```

The `exportSvg` function extracts computed styles from your styled elements and embeds them in the SVG, making it portable and self-contained.

<Warning>
Export functions require a browser environment. Always call them in event handlers or client-side code, not during SSR.
</Warning>

## Complete example

Here's a complete working example combining all the features:

```svelte App.svelte
<script>
  import { AsciiArt, exportSvg, exportSvgToPng } from 'svelte-asciiart';

  const text = `+----------+     _o<
|  Hello   |    \`\\,_
|  World!  |   (_)/ (_)
+----------+`;

  let svg = $state();

  async function copyToClipboard() {
    if (!svg) return;
    const markup = exportSvg(svg, {
      includeBackground: true,
      backgroundColor: '#f3f4f6'
    });
    await navigator.clipboard.writeText(markup);
    alert('SVG copied to clipboard!');
  }
</script>

<div class="container">
  <AsciiArt 
    bind:svg
    {text} 
    grid 
    frame 
    margin={[1, 2]} 
    cellAspect={0.6}
    gridClass="ascii-grid" 
    frameClass="ascii-frame" 
  />
  
  <button onclick={copyToClipboard}>
    Copy SVG to clipboard
  </button>
</div>

<style>
  .container {
    max-width: 600px;
    margin: 2rem auto;
  }
  
  :global(.ascii-grid) {
    stroke: #87cefa;
    stroke-width: 0.03;
    opacity: 0.5;
  }
  
  :global(.ascii-frame) {
    stroke: #ffb366;
    stroke-width: 0.05;
  }
  
  button {
    margin-top: 1rem;
    padding: 0.5rem 1rem;
    background: #ff3e00;
    color: white;
    border: none;
    border-radius: 0.25rem;
    cursor: pointer;
  }
  
  button:hover {
    background: #ff6b35;
  }
</style>
```
